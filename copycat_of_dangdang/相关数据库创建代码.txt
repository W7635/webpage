相关数据库创建代码：
1.users
CREATE TABLE `users` (
  `user_id` int NOT NULL AUTO_INCREMENT,
  `phone` varchar(20) CHARACTER SET utf8mb4 COLLATE utf8mb4_0900_ai_ci NOT NULL,
  `password` varchar(255) CHARACTER SET utf8mb4 COLLATE utf8mb4_0900_ai_ci NOT NULL,
  `status` tinyint(1) DEFAULT '1' COMMENT '0-禁用,1-激活',
  `user_type` tinyint(1) DEFAULT '2' COMMENT '1-管理员,2-普通会员',
  PRIMARY KEY (`user_id`)
) ENGINE=InnoDB AUTO_INCREMENT=7 DEFAULT CHARSET=utf8mb3

2.orders
CREATE TABLE `orders` (
  `order_id` int NOT NULL AUTO_INCREMENT,
  `user_phone` varchar(50) NOT NULL,
  `product_name` varchar(50) NOT NULL,
  `product_number` int DEFAULT NULL,
  `total_price` decimal(10,2) NOT NULL,
  `payment_status` varchar(20) DEFAULT '未支付' COMMENT '未支付,已支付',
  PRIMARY KEY (`order_id`),
  KEY `fk_orders_user` (`user_phone`)
) ENGINE=InnoDB AUTO_INCREMENT=14 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci

3.cart_products
CREATE TABLE `cart_products` (
  `product_id` int NOT NULL AUTO_INCREMENT,
  `product_name` varchar(100) NOT NULL,
  `price` decimal(10,2) NOT NULL,
  `number` int NOT NULL,
  `total_price` double NOT NULL,
  PRIMARY KEY (`product_id`)
) ENGINE=InnoDB AUTO_INCREMENT=13 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci
